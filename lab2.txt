% 1. Сформировать новый список из всех четных элементов исходного списка.

% если исходный список пуст, то результат - пустой
list_chet([], []).
     
% cписок четных элементов - объединение первого элемента исходного списка и
% списка четных элементов хвоста исходного списка,
% если первый элемент исходного списка кратен 2-м
list_chet([H | T], R) :-
    list_chet(T, TR),
    0 is mod(H, 2),
    R = [H | TR].

% cписок четных элементов - список четных элементов хвоста исходного списка,
% если первый элемент исходного списка не кратен 2-м
list_chet([H | T], R) :-
    list_chet(T, TR),
    1 is mod(H, 2),
    R = TR.
